"use strict";
function mapping(arr) {
let maps = new Map();
let numbers = [];
let booleans = [];
let strings = [];
for (let i = 0; i < arr.length; i++) {
    if (/\d+/.test(arr[i]) ) {
   numbers.push(arr[i]);
    maps.set("number", numbers);
  }  else if (/true|false/.test(arr[i])) {
   booleans.push(arr[i]);
    maps.set("boolean", booleans);
  }  else if (/\w+/.test(arr[i])) {
   strings.push(arr[i]);
    maps.set("string", strings);
  }
}
return maps;
}
mapping(['a', 1, 2, false, 'b']).forEach((value, key,) => { alert(`${key}: [${value}]`); });
